新建一个获取类型的函数
digoal=> create or replace function get_type (i_oid oid) returns name as $$
declare
v_typname name;
begin
select typname into v_typname from pg_type where oid=i_oid;
return v_typname;
end;
$$ language plpgsql;
CREATE FUNCTION
